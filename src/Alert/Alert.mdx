<Source dark language="jsx" code={`
<script lang="ts">
  import { Alert } from 'sveltestrap';

  const colors = [
    'primary',
    'secondary',
    'success',
    'danger',
    'warning',
    'info',
    'light',
    'dark'
  ];
</script>

{#each colors as color}
  <Alert {color}>
    <h4 class="alert-heading text-capitalize">{color}</h4>
    Lorem ipsum dolor sit amet, consectetur adipiscing elit.
    <a href="#todo" class="alert-link">
      Also, alert-links are colored to match
    </a>
    .
  </Alert>
{/each}
`} />

## Fade
<Canvas withSource="none">
  <Story of={AlertStories.Fade} />
</Canvas>

<p>You can disable the <code>Alert</code> fade transition by setting the <code>fade</code> prop to <code>false</code>.</p>

<Source dark language="jsx" code={`
<script lang="ts">
  import { Alert } from 'sveltestrap';

  let isOpen = true;

  const toggle = () => {
    visible = !visible;
  };
</script>

<Alert
  color="primary"
  {isOpen}
  {toggle}
  fade={false}
>
  I am a primary alert and I can be dismissed without animating!
</Alert>
`} />

## Controlled
<Canvas withSource="none">
  <Story of={AlertStories.Controlled} />
</Canvas>

<Source dark language="jsx" code={`
<script lang="ts">
  import { Alert, Button } from 'sveltestrap';

  let isOpen = true;

  const toggle = () => {
    visible = !visible;
  };
</script>

<Alert color="primary" isOpen={isOpen} toggle={() => (isOpen = false)}>
  I can be controlled via <code>isOpen</code> and <code>toggle</code>.
</Alert>

<Button on:click={toggle}>You can toggle me here.</Button>
`} />
